```python
# app/editor/chapters.py (PURE-часть)
MIN_CHAPTER_SEC = 2.0

def postprocess_chapters(raw: list[Chapter], duration: float) -> list[Chapter]:
    """Сортировка, кламп в [0,duration], выкидывание мусора, непрерывное покрытие."""
    valid = []
    for ch in raw:
        s = max(0.0, min(ch.start, duration))
        e = max(0.0, min(ch.end, duration))
        if e - s >= MIN_CHAPTER_SEC:
            valid.append(Chapter(start=s, end=e, title=ch.title.strip(), summary=ch.summary.strip()))
    valid.sort(key=lambda c: c.start)
    out: list[Chapter] = []
    for ch in valid:
        if out:
            prev = out[-1]
            if ch.start < prev.end:           # перекрытие → режем текущую
                ch = Chapter(start=prev.end, end=ch.end, title=ch.title, summary=ch.summary)
                if ch.end - ch.start < MIN_CHAPTER_SEC:
                    continue
            elif ch.start > prev.end:         # дыра → тянем предыдущую
                out[-1] = Chapter(start=prev.start, end=ch.start, title=prev.title, summary=prev.summary)
        out.append(ch)
    if out:
        out[-1] = Chapter(start=out[-1].start, end=duration, title=out[-1].title, summary=out[-1].summary)
    return out
```
